import React from 'react';
import { render, screen } from '@testing-library/react';
import Section from '../../components/templates/Section';
 
import * as DynamicTextsModule from '../../components/molecules/DynamicTexts';
import * as ImageAtomModule from '../../components/atoms/ImageAtom';
import * as CardsDisplayModule from '../../components/organisms/CardsDisplay';
 
describe('Section Component (con Jasmine)', () => {
  beforeEach(() => {
    spyOn(DynamicTextsModule, 'default').and.callFake((props) => (
      <div data-testid="DynamicTexts">{props.Texts}</div>
    ));
 
    spyOn(ImageAtomModule, 'default').and.callFake((props) => (
      <img data-testid="ImageAtom" src={props.src} alt={props.alt} />
    ));
 
    spyOn(CardsDisplayModule, 'default').and.callFake((props) => (
      <div data-testid="CardsDisplay">
        {props.content?.map((c, i) => (
          <span key={i}>{c.title}</span>
        ))}
      </div>
    ));
  });
 
  it('renderiza textos cuando el tipo es "text"', () => {
    const content = [{ type: 'text', text: 'Hola mundo', className: 'text-class' }];
    render(<Section content={content} />);
    expect(screen.getByTestId('DynamicTexts').textContent).toContain('Hola mundo');
  });
 
  it('renderiza una imagen cuando el tipo es "image"', () => {
    const content = [{ type: 'image', src: '/foto.png', alt: 'Foto de prueba', className: 'img-class' }];
    render(<Section content={content} />);
    const image = screen.getByTestId('ImageAtom');
    expect(image.getAttribute('src')).toBe('/foto.png');
    expect(image.getAttribute('alt')).toBe('Foto de prueba');
  });
 
  it('renderiza CardsDisplay cuando el tipo es "cards"', () => {
    const content = [{ type: 'cards', cards: [{ title: 'Card 1' }, { title: 'Card 2' }] }];
    render(<Section content={content} />);
    expect(screen.getByText('Card 1')).not.toBeNull();
    expect(screen.getByText('Card 2')).not.toBeNull();
  });
 
  it('renderiza CardsDisplay como lista cuando el tipo es "cardList"', () => {
    const content = [{ type: 'cardList', cards: [{ title: 'Card A' }] }];
    render(<Section content={content} />);
    expect(screen.getByText('Card A')).not.toBeNull();
  });
 
  it('aplica la clase pasada al contenedor', () => {
    const { container } = render(<Section content={[]} className="section-class" />);
    expect(container.firstChild.className).toContain('section-class');
  });
});
